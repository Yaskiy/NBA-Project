## 📝 Example: Using Page Objects

```typescript
import { test } from '@playwright/test';
import { GameNavigationPage } from '../src/pages/gameTeamNavigation';

test('Example using page objects', async ({ page }) => {
  const gamesPage = new GameNavigationPage(page);
  
  // Navigate and verify
  await gamesPage.navigateToGames();
  await gamesPage.verifyGamesPageLoaded();
  
  // Get data
  const gameCount = await gamesPage.getGameCount();
  console.log(`Found ${gameCount} games`);
  
  // Interact
  await gamesPage.clickNextDate();
  await gamesPage.toggleHideScores();
});
```
